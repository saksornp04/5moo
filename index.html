// --- Food class for falling objects --- class FallingFood { constructor(spawnX) { const template = FOOD_TEMPLATES[Math.floor(Math.random() * FOOD_TEMPLATES.length)]; this.name = template.name; this.group = template.group; this.emoji = template.emoji; this.size = 88; // ขยายวงกลมให้ใหญ่ขึ้น (เดิม 58) this.x = spawnX; this.y = -60; // เพิ่มความเร็วตกลงมาให้เร็วขึ้นชัดเจน (เดิม 0.6-1.1) this.speed = Math.random() * 1.4 + 1.8; this.wobbleSpeed = Math.random() * 0.02 + 0.01; this.wobbleAmt = Math.random() * 6 + 2; this.wobbleOffset = Math.random() * 100; this.isGrabbed = false; } update(time) { if (this.isGrabbed && currentPinchPoint) { this.x = currentPinchPoint.x; this.y = currentPinchPoint.y; } else { this.y += this.speed; this.x += Math.sin(time * this.wobbleSpeed + this.wobbleOffset) * 0.25; } } draw(ctx) { ctx.save(); ctx.shadowColor = 'rgba(255, 255, 255, 0.5)'; ctx.shadowBlur = this.isGrabbed ? 20 : 10; ctx.beginPath(); ctx.arc(this.x, this.y, this.size / 2, 0, Math.PI * 2); ctx.fillStyle = this.isGrabbed ? "rgba(16, 185, 129, 0.9)" : "rgba(30, 41, 59, 0.9)"; ctx.fill(); ctx.strokeStyle = this.isGrabbed ? "#10b981" : "#ffffff"; ctx.lineWidth = 3.5; ctx.stroke(); // วาดอิโมจิ (ขยายใหญ่ขึ้นชัดเจน) ctx.shadowBlur = 0; ctx.font = `${this.size * 0.62}px Arial`; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.fillText(this.emoji, this.x, this.y - 8); // วาดชื่ออาหาร (ตัวหนา ใหญ่ขึ้น มีขอบดำให้อ่านง่าย) ctx.font = 'bold 15px Mitr'; ctx.lineWidth = 4; ctx.strokeStyle = 'rgba(0, 0, 0, 0.85)'; ctx.strokeText(this.name, this.x, this.y + 22); ctx.fillStyle = '#ffffff'; ctx.fillText(this.name, this.x, this.y + 22); ctx.restore(); } }
